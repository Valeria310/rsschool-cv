function findOutlier(integers) {
  if (integers[0] % 2 == 0 &&
 integers[1] % 2 == 0) {
    for (i = 0; i < integers.length; i++) {
      if (integers[i] % 2 != 0) {
        return integers[i];
      }
    }
  } else if (integers[0] % 2 != 0 &&
 integers[1] % 2 != 0) {
    for (i = 0; i < integers.length; i++) {
      if (integers[i] % 2 == 0) {
        return integers[i];
      }
    }
  } else if (
    (integers[0] % 2 == 0 &&
 integers[2] % 2 == 0) ||
    (integers[1] % 2 == 0 &&
 integers[2] % 2 == 0)
  ) {
    if (integers[0] % 2 != 0 ||
 integers[0] % 2 != 0) {
      return integers[0];
    } else {
      return integers[1];
    }
